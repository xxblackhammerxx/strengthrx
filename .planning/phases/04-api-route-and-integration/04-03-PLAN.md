Create new file src/app/api/admin/retry-pb-sync/route.ts with a POST handler:

```typescript
import { NextRequest, NextResponse } from 'next/server'
import { getPayload } from 'payload'
import config from '@payload-config'
import { headers as nextHeaders } from 'next/headers'
import { createPracticeBetterClient } from '@/lib/practice-better'

export async function POST(request: NextRequest) {
  try {
    // 1. Authenticate admin
    const payload = await getPayload({ config })
    const headers = await nextHeaders()
    const { user } = await payload.auth({ headers })

    if (!user || (user as any).collection !== 'admins') {
      return NextResponse.json({ error: 'Unauthorized' }, { status: 401 })
    }

    // 2. Parse body
    const body = await request.json()
    const { clientId } = body

    if (!clientId) {
      return NextResponse.json({ error: 'clientId is required' }, { status: 400 })
    }

    // 3. Fetch client record
    const client = await payload.findByID({
      collection: 'clients',
      id: clientId,
    })

    if (!client) {
      return NextResponse.json({ error: 'Client not found' }, { status: 404 })
    }

    if (client.practiceBetterSyncStatus !== 'failed') {
      return NextResponse.json({ error: 'Client sync status is not failed' }, { status: 400 })
    }

    // 4. If client already has a PB ID (partial success), skip creation and just mark synced
    if (client.practiceBetterId) {
      await payload.update({
        collection: 'clients',
        id: clientId,
        data: { practiceBetterSyncStatus: 'synced' },
      })
      return NextResponse.json({ success: true, alreadyExisted: true })
    }

    // 5. Call Practice Better
    const pbResult = await createPracticeBetterClient({
      firstName: client.firstName,
      lastName: client.lastName,
      email: client.email,
    })

    // 6. Update client with PB ID and synced status
    await payload.update({
      collection: 'clients',
      id: clientId,
      data: {
        practiceBetterId: pbResult.id,
        practiceBetterSyncStatus: 'synced',
      },
    })

    return NextResponse.json({ success: true })
  } catch (error) {
    console.error('PB retry sync error:', error)

    // On PB failure, keep status as failed
    return NextResponse.json(
      { error: 'Failed to sync with Practice Better. Please try again.' },
      { status: 500 },
    )
  }
}
```

Key design decisions:
- Inline admin auth check (not using getAuthenticatedAdmin because it uses next/headers for Server Components -- this is a Route Handler, which gets headers from the request). Actually, read auth.ts again -- getAuthenticatedAdmin DOES work in Route Handlers since it calls `await nextHeaders()` internally. However, for Route Handlers we have the request object, and getAuthenticatedAdmin is designed for Server Components. To be safe and consistent with how other API routes work (none of them use getAuthenticatedAdmin), use the inline `payload.auth({ headers })` pattern shown above.
- Checks practiceBetterSyncStatus === 'failed' before allowing retry
- Handles the "already has PB ID" edge case (Pitfall 7 from research) by skipping creation
- This is a BLOCKING call (not fire-and-forget) because the admin is waiting for the result

   ```tsx
   function FailedSyncRow({ client }: { client: FailedPbSyncClient }) {
     const [status, setStatus] = useState<'idle' | 'retrying' | 'success' | 'error'>('idle')

     const handleRetry = async () => {
       setStatus('retrying')
       try {
         const res = await fetch('/api/admin/retry-pb-sync', {
           method: 'POST',
           headers: { 'Content-Type': 'application/json' },
           body: JSON.stringify({ clientId: client.id }),
         })
         if (res.ok) {
           setStatus('success')
         } else {
           setStatus('error')
         }
       } catch {
         setStatus('error')
       }
     }

     if (status === 'success') {
       return (
         <TableRow className="bg-green-50">
           <TableCell className="font-medium">{client.name}</TableCell>
           <TableCell>{client.email}</TableCell>
           <TableCell>{new Date(client.createdAt).toLocaleDateString('en-US', { month: 'short', day: 'numeric', year: 'numeric' })}</TableCell>
           <TableCell><span className="text-sm text-green-600 font-medium">Synced</span></TableCell>
         </TableRow>
       )
     }

     return (
       <TableRow>
         <TableCell className="font-medium">{client.name}</TableCell>
         <TableCell>{client.email}</TableCell>
         <TableCell>{new Date(client.createdAt).toLocaleDateString('en-US', { month: 'short', day: 'numeric', year: 'numeric' })}</TableCell>
         <TableCell>
           <button
             onClick={handleRetry}
             disabled={status === 'retrying'}
             className="text-sm font-medium text-blue-600 hover:underline disabled:opacity-50 disabled:no-underline"
           >
             {status === 'retrying' ? 'Retrying...' : status === 'error' ? 'Retry Again' : 'Retry Sync'}
           </button>
         </TableCell>
       </TableRow>
     )
   }
   ```
